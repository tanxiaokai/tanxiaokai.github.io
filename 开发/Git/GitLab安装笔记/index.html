GitLab安装维护笔记

发表于 2018-01-13 | 分类于 开发 ， Git

​ GitLab 是一个用于仓库管理系统的开源项目。使用Git作为代码管理工具，并在此基础上搭建起来的web服务。可通过Web界面进行访问公开的或者私人项目。它拥有与Github类似的功能，能够浏览源代码，管理缺陷和注释。可以管理团队对仓库的访问，它非常易于浏览提交过的版本并提供一个文件历史库。团队成员可以利用内置的简单聊天程序(Wall)进行交流。它还提供一个代码片段收集功能可以轻松实现代码复用。

二、安装

  • 前期准备

    #使用国内镜像：
#新建 vi /etc/yum.repos.d/gitlab-ce.repo，内容为：

[gitlab-ce]  
name=gitlab-ce  
baseurl=http://mirrors.tuna.tsinghua.edu.cn/gitlab-ce/yum/el7  
repo_gpgcheck=0  
gpgcheck=0  
enabled=1  
gpgkey=https://packages.gitlab.com/gpg.key  

sudo yum makecache

  • 安装

    #安装ssh服务，设置防火墙
sudo yum install -y curl policycoreutils-python openssh-server
sudo systemctl enable sshd
sudo systemctl start sshd
sudo firewall-cmd --permanent --add-service=http
sudo systemctl reload firewalld

#邮件服务
sudo yum install postfix -y
sudo systemctl enable postfix
sudo systemctl start postfix

#安装gitlab
curl https://packages.gitlab.com/install/repositories/gitlab/gitlab-ee/script.rpm.sh | sudo bash
sudo yum install gitlab-ce -y

  • 汉化

    mkdir /opt/tmp
cd /opt/tmp

#下载最新的汉化包
#git clone https://gitlab.com/xhang/gitlab.git
#指定版本下载,如下载10.5.4，可以运行下面的语句
git clone https://gitlab.com/xhang/gitlab.git -b v10.5.4-zh

#查看该汉化补丁的版本
cat gitlab/VERSION

#比较汉化标签和原标签，导出 patch 用的 diff 文件
cd /opt/tmp/gitlab/
git diff v10.5.4 v10.5.4-zh > ../10.5.4-zh.diff

#将补丁更新到gitlab中
cd /opt/tmp
yum install patch -y 
patch -d /opt/gitlab/embedded/service/gitlab-rails -p1 < 10.5.4-zh.diff

cd /opt
rm -rf /opt/tmp

三、配置

  • 配置gitlab

    vi /etc/gitlab/gitlab.rb

修改:
external_url 'http://120.77.2.93:7000'

unicorn['listen'] = '127.0.0.1'
unicorn['port'] = 3000

#主目录
git_data_dirs({
   "default" => {
     "path" => "/home/git/gitlab-repositories"
    }
 })

#邮件服务器设置 
gitlab_rails['smtp_enable'] = true
gitlab_rails['smtp_address'] = "smtp.exmail.qq.com"
gitlab_rails['smtp_port'] = 465
gitlab_rails['smtp_user_name'] = "dev@1ycloud.net"
gitlab_rails['smtp_password'] = "aokA1234"
#gitlab_rails['smtp_domain'] = "exmail.qq.com"
gitlab_rails['smtp_authentication'] = "login"
gitlab_rails['smtp_enable_starttls_auto'] = true
gitlab_rails['smtp_tls'] = true
gitlab_rails['gitlab_email_from'] = 'dev@1ycloud.net'

user['git_user_name'] = "GitLab"
user['git_user_email'] = "dev@1ycloud.net"

  • 启动/维护

     sudo gitlab-ctl reconfigure  #重新加载配置, 每次重新配置，都需要执行
 sodu gitlab-ctl stop     #停止
 sodu gitlab-ctl start    #启动

 查看日志:
 tail -f /var/log/gitlab/gitlab-rails/production.log

 切换git注目录
 vi /etc/passwd   #找git用户，有两个1个是老的git服务，一个是gitlab服务，开启需要的即可

 #发送邮件测试
 gitlab-rails console     
 Notify.test_email('tank@1ycloud.net', 'Message Subject', 'Message Body').deliver_now

四、导入项目

#创建导入目录 
#sudo -u git mkdir /var/opt/gitlab/git-data/repository-import-<群组名>/群组名
#示例
sudo -u git mkdir /home/git/gitlab-repositories/repository-import-other
sudo -u git mkdir /home/git/gitlab-repositories/repository-import-other/other

#拷贝原来Git版本库
cp -rf /home/git/repositories/bm_radio.git /home/git/gitlab-repositories/repository-import-other/other/

#修改权限
chown -R git:git  /home/git/gitlab-repositories/repository-import-other

#导入代码
gitlab-rake gitlab:import:repos['/home/git/gitlab-repositories/repository-import-other']
