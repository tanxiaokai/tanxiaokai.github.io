rabbitmq常用操作

发表于 2018-03-16 | 分类于 开发 ， RabbitMq

rabbitmqadmin使用

配置rabbitmqadmin

   #查找位置
   find /var/lib/rabbitmq/ -name "rabbitmqadmin"

   #复制，路径更加查找的结果来
   cp -a /var/lib/rabbitmq/rabbit@*-plugins-expand/.../rabbitmqadmin /usr/bin/rabbitmqadmin   

   #添加执行权限
   chmod +x /usr/bin/rabbitmqadmin

使用rabbitmqadmin

查看exchanges

​```
rabbitmqadmin -u rabbitmq_admin -p 'passwd' -V daxiang list exchanges
​```

查看队列

   rabbitmqadmin -u rabbitmq_admin -p 'passwd' -V daxiang list queues

新增队列

   rabbitmqadmin -u rabbitmq_admin -p 'passwd' -V daxiang declare queue name=settlement_biz_queue durable=true

查看队列中的消息

   rabbitmqadmin -u rabbitmq_admin -p 'passwd' -V prod get queue=eis_queue requeue=true  count=106

发布消息

   rabbitmqadmin -u rabbitmq_admin -p 'passwd' -V prod publish exchange=ak.topic routing_key=spd_biz properties='{"headers":{"msgType":"新增企业产品"},"content_type":"text/plain"}' payload='{"edsGoodsId":"ff8080815e409cfe015e50c52924001d","userId":"ff8080815d54b5c4015d54d5acf80061"}'

清除队列消息

   rabbitmqadmin --host=120.77.2.93 -u rabbitmq_admin -p 'passwd' -V ittest purge queue name=batch_task

新增binding

   rabbitmqadmin -u rabbitmq_admin -p 'passwd' -V daxiang declare binding source="ak.topic" destination_type="queue" destination="settlement_biz_queue" routing_key="spd_biz"

删除binding

​```
rabbitmqadmin -u rabbitmq_admin -p 'passwd' -V prod delete binding source="amq.topic" destination_type="queue" destination="batch_task" properties_key="create_batch_task"
​```

使用示例

   #!/bin/bash
   #该脚本用于初始化新用户的RabbitMq相关环境

   #参数1是新用户对应的vHost名称
   vHost=$1

   #RabbitMq管理员用户名和密码
   adminName=admin
   password='123456'
   topicName='ak.topic'

   echo "初始化[${vHost}]数据"

   #删除vHost
   rabbitmqadmin -u $adminName -p $password delete vhost name=${vHost}

   #创建vHost
   rabbitmqadmin -u $adminName -p $password declare vhost name=${vHost}

   #赋予权限
   rabbitmqadmin -u $adminName -p $password declare permission vhost=${vHost} user=$adminName configure=".*" write=".*" read=".*"

   #创建Exchanges
   rabbitmqadmin -u $adminName -p $password  -V $vHost declare exchange name=$topicName type=topic

   #创建queue
   rabbitmqadmin -u $adminName -p $password -V $vHost declare queue name=batch_task durable=true
   rabbitmqadmin -u $adminName -p $password -V $vHost declare queue name=eis_queue durable=true
   rabbitmqadmin -u $adminName -p $password -V $vHost declare queue name=settlement_biz_queue durable=true
   rabbitmqadmin -u $adminName -p $password -V $vHost declare queue name=spd_biz_queue durable=true

   #创建队列绑定关系
   rabbitmqadmin -u $adminName -p $password -V $vHost declare binding source=$topicName destination_type="queue" destination="batch_task" routing_key="create_batch_task"
   rabbitmqadmin -u $adminName -p $password -V $vHost declare binding source=$topicName destination_type="queue" destination="eis_queue" routing_key="sync_task"
   rabbitmqadmin -u $adminName -p $password -V $vHost declare binding source=$topicName destination_type="queue" destination="settlement_biz_queue" routing_key="spd_biz"
   rabbitmqadmin -u $adminName -p $password -V $vHost declare binding source=$topicName destination_type="queue" destination="spd_biz_queue" routing_key="spd_biz"

   echo "初始化[${vHost}]数据-完成"

   #显示vhosts
   echo "------------------------vhosts-------------------------------"
   rabbitmqadmin -u $adminName -p $password -V $vHost list vhosts

   #显示exchanges
   echo "------------------------exchanges-------------------------------"
   rabbitmqadmin -u $adminName -p $password -V $vHost list exchanges

   #显示queues
   echo "-------------------------queues---------------------------------"
   rabbitmqadmin -u $adminName -p $password -V $vHost list queues

   #显示bindings
   echo "--------------------------bindings------------------------------"
   rabbitmqadmin -u $adminName -p $password -V $vHost list bindings
